爬取网页

def getData(baseurl):
datalist = [] # 用来存储爬取的网页信息
for i in range(0, 10): # 调用获取页面信息的函数，10次
url = baseurl + str(i * 25)
print(f”正在爬取第{i+1}页：{url}”)
html = askURL(url) # 保存获取到的网页源码

    if not html:  # 如果获取不到网页内容，跳过
        print(f"第{i+1}页爬取失败，跳过")
        continue
        
    # 2.逐一解析数据
    soup = BeautifulSoup(html, "html.parser")
    items = soup.find_all('div', class_="item")  # 查找符合要求的字符串
    
    if not items:  # 如果没有找到电影条目，跳过
        print(f"第{i+1}页没有找到电影条目，跳过")
        continue
        
    for item in items:
        data = []  # 保存一部电影所有信息
        item_str = str(item)  # 避免变量名冲突
        
        # 解析电影详情链接
        link_list = re.findall(findLink, item_str)
        link = link_list[0] if link_list else ""
        data.append(link)
        
        # 解析图片链接
        imgSrc_list = re.findall(findImgSrc, item_str)
        imgSrc = imgSrc_list[0] if imgSrc_list else ""
        data.append(imgSrc)
        
        # 解析影片标题
        titles = re.findall(findTitle, item_str)
        if len(titles) == 2:
            ctitle = titles[0].strip()
            data.append(ctitle)
            otitle = titles[1].replace("/", "").strip()  # 消除转义字符
            data.append(otitle)
        elif len(titles) == 1:
            data.append(titles[0].strip())
            data.append(' ')
        else:
            data.append(' ')
            data.append(' ')
        
        # 解析评分
        rating_list = re.findall(findRating, item_str)
        rating = rating_list[0] if rating_list else "0"
        data.append(rating)
        
        # 解析评价数
        judgeNum_list = re.findall(findJudge, item_str)
        judgeNum = judgeNum_list[0] if judgeNum_list else "0"
        data.append(judgeNum)
        
        # 解析概况
        inq_list = re.findall(findInq, item_str)
        if inq_list:
            inq = inq_list[0].replace("。", "").strip()
            data.append(inq)
        else:
            data.append(" ")
        
        # 解析相关信息（修复这里的IndexError）
        bd_list = re.findall(findBd, item_str)
        if bd_list:
            bd = bd_list[0]
            bd = re.sub('<br(\s+)?/>(\s+)?', "", bd)  # 去除br标签
            bd = re.sub('/', "", bd)  # 去除/符号
            bd = re.sub('\s+', ' ', bd)  # 多个空格替换为单个空格
            data.append(bd.strip())
        else:
            data.append(" ")
        
        # 只有当数据完整时才添加（至少有中文名和评分）
        if data[2] and data[4] != "0":
            datalist.append(data)

print(f"共爬取到{len(datalist)}部电影数据")
return datalist

得到指定一个URL的网页内容

def askURL(url):
head = { # 模拟浏览器头部信息，向豆瓣服务器发送消息
“User-Agent”: “Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36”,
“Accept”: “text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,/;q=0.8”,
“Accept-Language”: “zh-CN,zh;q=0.9,en;q=0.8”,
“Referer”: “https://movie.douban.com/"
}
# 用户代理，表示告诉豆瓣服务器，我们是什么类型的机器、浏览器

request = urllib.request.Request(url, headers=head)
html = ""
try:
    response = urllib.request.urlopen(request, timeout=30)  # 添加超时设置
    html = response.read().decode("utf-8")
except urllib.error.URLError as e:
    if hasattr(e, "code"):
        print(f"HTTP错误代码：{e.code}")
    if hasattr(e, "reason"):
        print(f"错误原因：{e.reason}")
except Exception as e:
    print(f"获取网页内容时发生错误：{str(e)}")
return html

保存数据到表格

def saveData(datalist, savepath):
print(“正在保存数据…”)
try:
book = xlwt.Workbook(encoding=”utf-8”, style_compression=0) # 创建workbook对象
sheet = book.add_sheet(‘豆瓣电影Top250’, cell_overwrite_ok=True) # 创建工作表
col = (“电影详情链接”, “图片链接”, “影片中文名”, “影片外国名”, “评分”, “评价数”, “概况”, “相关信息”)

    # 写入列名
    for i in range(0, 8):
        sheet.write(0, i, col[i])
    
    # 写入数据
    for i in range(len(datalist)):
        data = datalist[i]
        for j in range(0, 8):
            # 确保数据不会超出单元格限制
            if j < len(data):
                sheet.write(i + 1, j, data[j][:32767] if isinstance(data[j], str) else data[j])  # 限制字符串长度
            else:
                sheet.write(i + 1, j, "")
    
    book.save(savepath)
    print(f"数据已成功保存到：{savepath}")
except Exception as e:
    print(f"保存数据时发生错误：{str(e)}")
